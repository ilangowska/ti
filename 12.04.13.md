```c
#include <stdio.h>
double potega(double a, int n);
int main() {
int n;
double a, w;
printf("podaj liczbe a: \a");
scanf("%lf" , &a);
printf("do jakiej potegi chcesz podniesc: ");
scanf("%d" , &n);
w=potega(a,n);
printf("Wartosc  a^n=%lf " , w);
getchar (),getchar ();
return 0;
}
double potega(double a, int n){
int k=0;
double iloczyn=1;
while (k<n) {
      iloczyn=iloczyn*a;
k++;
}
return iloczyn;
}
```
